% Points
\foreach \p/\l in {(-2.0,1.5)/P_1,(-1.0,0.8)/P_2,(0.5,1.6)/P_3,(1.5,0.3)/P_4,(2.2,1.2)/P_5} {

\node[font=\footnotesize,red!70!black] at (2.2,0.5) {$P_n$};

\node[font=\bfseries] at (7.4,3.3) {(b) Spherical Vector Encoding (SPSO)};
\node[font=\small] at (7.4,2.7) {segments = $(\rho_j, \psi_j, \phi_j)$};

% Draw a UAV and path in spherical representation
% UAV icon (simple)
\node[font=\Large] at (5.2,2.2) {\scriptsize UAV};

% ===== Bottom: conversion formula =====
\node[font=\small,align=center] at (3.7,-3.2) {$\begin{aligned}
    x_{j+1} &= x_j + \rho_j \sin\psi_j \cos\phi_j \\
    y_{j+1} &= y_j + \rho_j \sin\psi_j \sin\phi_j \\
    z_{j+1} &= z_j + \rho_j \cos\psi_j
\end{aligned}$};
